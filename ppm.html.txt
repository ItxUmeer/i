<h2>PPM Records</h2>

<input id="equipment" placeholder="Equipment">
<input id="date" type="date">
<input id="technician" placeholder="Technician">

<button onclick="addPPM()">Add PPM</button>

<ul id="ppmList"></ul>

<script type="module">
import { db } from './firebase.js';
import { collection, addDoc, getDocs } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js";

window.addPPM = async function() {
  await addDoc(collection(db, "ppm"), {
    equipment: equipment.value,
    date: date.value,
    technician: technician.value
  });
  alert("Saved");
  loadPPM();
}

async function loadPPM() {
  const list = document.getElementById("ppmList");
  list.innerHTML = "";

  const querySnapshot = await getDocs(collection(db, "ppm"));
  querySnapshot.forEach(doc => {
    let li = document.createElement("li");
    li.textContent = doc.data().equipment + " - " + doc.data().date;
    list.appendChild(li);
  });
}

loadPPM();
</script>
